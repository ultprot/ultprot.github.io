# BOJ 곱셈
BOJ 1629 곱셈


[문제](https://www.acmicpc.net/problem/1629)


## n을 log n으로 줄일 수 있을 때는 줄여야 한다.


## C++
```cpp
#include <iostream>
using namespace std;

class Problem {
private:
	const long long a;
	const long long b;
	const long long c;
public:
	Problem(long long a, long long b, long long c):a(a), b(b), c(c) {
	}
	long long getAnswer(long long cb) {
		long long temp = 0;
		if (cb == 1) {
			temp = a;
		}
		else {
			temp = getAnswer(cb / 2);
			temp *= temp;
			temp %= c;
			if (cb % 2 != 0) {
				temp *= (a % c);
			}
		}
		return temp % c;
	}
	long long getAnswer() {
		return getAnswer(b);
	}
};

int main(void) {
	long long a, b, c;
	cin >> a >> b >> c;
	Problem problem(a, b, c);
	cout << problem.getAnswer() << endl;;
}
```

class Problem{
    constructor(a, b, c){
        this.a = BigInt(a);
        this.b = Number(b);
        this.c = BigInt(c);
    }

    getAnswer(cb){
        let temp = BigInt(0);
        if(cb == 1){
            temp = this.a;
        }else{
            temp = this.getAnswer(parseInt(cb/2));
            temp *= temp;
            temp %= this.c;
            if(cb%2 !== 0){
                temp *= (this.a % this.c);
            }
        }
        return temp % this.c;
    }
}
