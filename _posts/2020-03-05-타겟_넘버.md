## C++
```cpp
#include <vector>

using namespace std;

class Problem {
private:
	vector<int>& numbers;
	int target;
	int answer;
	const int last;
public:
	Problem(vector<int>& numbers, int target)
		:numbers(numbers), 
		target(target),
		last(numbers.size())
	{
		this->answer = 0;
	}
	void newRound(int round, int value) {
		if (round == this->last) {
            if(value == this->target){
                this->answer++;
            }
		}
		else {
            int current = this->numbers[round];
            int plus = value + current;
            int minus = value - current;
			round++;
			newRound(round, plus);
			newRound(round, minus);
		}
        return;
	}
	int getAnswer() {
		newRound(0, 0);
		return this->answer;
	}
};

int solution(vector<int> numbers, int target) {
    int answer = 0;
    Problem problem(numbers, target);
	answer = problem.getAnswer();
    return answer;
}
```
