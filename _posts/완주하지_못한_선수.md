# 완주하지 못한 선수
## C++
- 두 배열 정렬하고 루프를 돌며 비교
```cpp
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

string solution(
    vector<string> participant, 
    vector<string> completion
)
{
    sort(participant.begin(), participant.end());
    sort(completion.begin(), completion.end());
    string answer = "";
    int end = participant.size();
    for(int i=0;i<end;i++){
        if(i==end-1){
            answer = participant[i];
            break;
        }
        if(participant[i]!=completion[i]){
            answer = participant[i];
            break;
        }
    }
    return answer;
}
```
